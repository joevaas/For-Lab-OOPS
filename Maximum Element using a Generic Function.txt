Start the program.

Call the generic method with different data types (e.g., Integer, Double, String).

public class GenericMaxFinder {

    // Step 1: Generic method to find maximum
    public static <T extends Comparable<T>> T findMax(T a, T b, T c) {
        T max = a;
        if (b.compareTo(max) > 0)
            max = b;
        if (c.compareTo(max) > 0)
            max = c;
        return max;
    }

    public static void main(String[] args) {
        System.out.println("Maximum of 10, 25, 17 is: " + findMax(10, 25, 17));
        System.out.println("Maximum of 5.4, 2.8, 9.1 is: " + findMax(5.4, 2.8, 9.1));
        System.out.println("Maximum of 'Apple', 'Banana', 'Mango' is: " + findMax("Apple", "Banana", "Mango"));
    }
}

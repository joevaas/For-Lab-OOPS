// Mini Project: Traffic Light Simulation

class TrafficLight extends Thread {
    private String[] lights = {"RED", "YELLOW", "GREEN"};

    public void run() {
        try {
            while (true) {
                for (String light : lights) {
                    switch (light) {
                        case "RED":
                            System.out.println("RED Light - STOP üö¶");
                            Thread.sleep(3000); // Wait 3 seconds
                            break;

                        case "YELLOW":
                            System.out.println("YELLOW Light - READY ‚ö†Ô∏è");
                            Thread.sleep(2000); // Wait 2 seconds
                            break;

                        case "GREEN":
                            System.out.println("GREEN Light - GO ‚úÖ");
                            Thread.sleep(3000); // Wait 3 seconds
                            break;
                    }
                    System.out.println("-----------------------------");
                }
            }
        } catch (InterruptedException e) {
            System.out.println("Traffic Light simulation interrupted.");
        }
    }
}

// Main Class
public class TrafficLightSimulation {
    public static void main(String[] args) {
        TrafficLight t = new TrafficLight();
        t.start();  // Start simulation
    }
}
